\usepackage{xcolor}
\usepackage{listings}

% Color definitions
\definecolor{JSON-value}{RGB}{42,0.0,255}
\definecolor{JSON-element}{RGB}{127,0,85}
\definecolor{JSON-comment}{rgb}{0,0.5,0}
\colorlet{JSON-number}{magenta!60!black}
\definecolor{JSON-obj-brace}{RGB}{20,105,176}
\definecolor{JSON-lst-brace}{RGB}{20,105,176}
\definecolor{JSON-background}{HTML}{EEEEEE}

% JSON HIGHLIGHTING VERSION 1:
\lstdefinelanguage{json}{
    % language grammar
    string=[b]",
    comment=[l]{\%},
    keywords={true, false, null},
    % formatting and styling
    basicstyle=\normalfont\ttfamily,
    breakatwhitespace=false,         
    breaklines=true,                 
    captionpos=b,                    
    keepspaces=true,                 
    numbers=left,                    
    numbersep=8pt,                  
    showspaces=false,                
    showstringspaces=false,
    showtabs=false,                  
    tabsize=2,
    numberstyle=\scriptsize,
    stepnumber=1,
    frame=lines,
    % colors and other styles
    moredelim=[is][\color{JSON-value}]{<@}{@>}, % important string values must have <@, and @> as delimiters; format them like: <@"<value>"@>  
    backgroundcolor=\color{JSON-background},   
    commentstyle=\color{JSON-comment},
    keywordstyle=\color{JSON-value},
    stringstyle=\color{JSON-element},
    literate=
        *{0}{{{\color{JSON-number}0}}}{1}
        {1}{{{\color{JSON-number}1}}}{1}
        {2}{{{\color{JSON-number}2}}}{1}
        {3}{{{\color{JSON-number}3}}}{1}
        {4}{{{\color{JSON-number}4}}}{1}
        {5}{{{\color{JSON-number}5}}}{1}
        {6}{{{\color{JSON-number}6}}}{1}
        {7}{{{\color{JSON-number}7}}}{1}
        {8}{{{\color{JSON-number}8}}}{1}
        {9}{{{\color{JSON-number}9}}}{1}
        {:}{{{\color{punct}{:}}}}{1}
        {,}{{{\color{punct}{,}}}}{1}
        {\{}{{{\color{JSON-obj-brace}{\{}}}}{1}
        {\}}{{{\color{JSON-obj-brace}{\}}}}}{1}
        {[}{{{\color{JSON-lst-brace}{[}}}}{1}
        {]}{{{\color{JSON-lst-brace}{]}}}}{1},
}
